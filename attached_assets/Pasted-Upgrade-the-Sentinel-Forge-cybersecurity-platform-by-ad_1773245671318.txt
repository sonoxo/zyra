from sqlalchemy import Column, Integer, String
from backend.database import Base

class Incident(Base):

    __tablename__ = "incidents"

    id = Column(Integer, primary_key=True)
    title = Column(String)
    severity = Column(String)
    status = Column(String)
    assigned_to = Column(String)

class Vulnerability(Base):

    __tablename__ = "vulnerabilities"

    id = Column(Integer, primary_key=True)
    title = Column(String)
    severity = Column(String)
    status = Column(String)

class Risk(Base):

    __tablename__ = "risks"

    id = Column(Integer, primary_key=True)
    description = Column(String)
    likelihood = Column(Integer)
    impact = Column(Integer)
    owner = Column(String)

MODULE 4: SECRETS SCANNING

backend/modules/secrets_scanner.py

import re

patterns = [
    r"AKIA[0-9A-Z]{16}",
    r"AIza[0-9A-Za-z\\-_]{35}"
]

def scan_for_secrets(code):

    findings = []

    for pattern in patterns:

        matches = re.findall(pattern, code)

        for m in matches:

            findings.append({
                "secret": m,
                "severity": "critical"
            })

    return findings

API keys
tokens
passwords
private keys

import socket

def discover_ports(host):

    open_ports = []

    for port in [80,443,22,8080]:

        s = socket.socket()

        try:
            s.connect((host, port))
            open_ports.append(port)
        except:
            pass

    return open_ports

Track:

subdomains
IP ranges
open ports
certificate expiration
exposed services

MODULE 7: SECURITY POSTURE SCORE TRENDING

backend/modules/posture_score.py

def calculate_security_score(vulns, incidents):

    score = 100

    score -= vulns * 5
    score -= incidents * 3

    return score

backend/main.py

from fastapi import FastAPI
from backend.modules.incident_response import create_incident
from backend.modules.vulnerability_lifecycle import create_vulnerability
from backend.modules.posture_score import calculate_security_score

app = FastAPI(title="Sentinel Forge")

@app.get("/")
def home():

    return {
        "platform":"Sentinel Forge",
        "modules":[
            "Incident Response",
            "Vulnerability Lifecycle",
            "SBOM Supply Chain",
            "Secrets Scanning",
            "Risk Register",
            "Attack Surface",
            "Security Posture"
        ]
    }

@app.post("/incident")

def new_incident(title:str,severity:str):

    return create_incident(title,severity)

@app.post("/vulnerability")

def new_vulnerability(issue:str):

    return create_vulnerability(issue)

@app.get("/security-score")

def security_score(vulns:int,incidents:int):

    return {
        "score":calculate_security_score(vulns,incidents)
    }
